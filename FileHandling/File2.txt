function calcSteps(sequence) {
    console.log(sequence);

    const steps = [];

    for (let i = 0; i < sequence.length - 1; i++) {
        console.log(sequence);
        console.log(`First value is ${sequence[i]}`);
        console.log(`Second Value is ${sequence[i + 1]}`);

        let step = sequence[i + 1] - sequence[i];

        // Check if step is numeric and not NaN before pushing to steps
        typeof step === 'number' && !isNaN(step) ? steps.push(step) : null;

        console.log(steps);
        console.log("********************");
    }

    // Handle the last element outside the loop
    let lastStep = sequence[sequence.length - 1];
    typeof lastStep === 'number' && !isNaN(lastStep) ? steps.push(lastStep) : null;

    console.log(steps);
}
